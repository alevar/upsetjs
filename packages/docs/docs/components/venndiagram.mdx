---
title: Venn Diagram
---

For comparison and convenience reasons UpSet.js also has a Venn Diagram component for rendering two or three sets.

```jsx live
// import React from 'react';

function Default() {
  return <VennDiagram sets={GameOfThronesDataSet.sets} width={780} height={500} />;
}
```

## Skeleton

```jsx live
// import React from 'react';

function Skeleton() {
  return <VennDiagramSkeleton width={780} height={500} />;
}
```

## Title

```jsx live
// import React from 'react';

function Title() {
  return (
    <VennDiagram
      sets={GameOfThronesDataSet.sets}
      width={780}
      height={500}
      title="Game of Thrones"
      description="Information about the characters of the tv series Game of Thrones"
    />
  );
}
```

## Interactivity

```jsx live
// import React from 'react';

function Interactivity() {
  const [selection, setSelection] = React.useState(null);
  return (
    <VennDiagram
      sets={GameOfThronesDataSet.sets}
      width={780}
      height={500}
      selection={selection}
      onHover={setSelection}
    />
  );
}
```

function Click() {

## Small Sets

```jsx live
// import React from 'react';

function SmallSets() {
  return (
    <div>
      <VennDiagram
        width={780}
        height={500}
        sets={asSets([
          { name: 'one', elems: [1, 2, 3, 5, 7, 8, 11, 12, 13] },
          { name: 'two', elems: [1, 2, 4, 5, 10] },
          { name: 'three', elems: [1, 5, 6, 7, 8, 9, 10, 12, 13] },
        ])}
      />
      <VennDiagram
        width={780}
        height={500}
        sets={asSets([
          { name: 'one', elems: [1, 2, 3] },
          { name: 'two', elems: [1, 2, 4, 5] },
          { name: 'three', elems: [1, 5] },
        ])}
      />
    </div>
  );
}
```
