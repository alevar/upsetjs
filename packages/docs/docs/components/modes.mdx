## Default

```jsx live
// import React from 'react';

function Example() {
  const [selection, setSelection] = React.useState(null);
  const sets = GameOfThronesDataSet.sets;
  const combinations = useMemo(
    () => ({
      type: 'intersection',
    }),
    []
  );
  return (
    <UpSetJS
      sets={sets}
      width={780}
      height={400}
      selection={selection}
      onHover={setSelection}
      combinations={combinations}
    />
  );
}
```
