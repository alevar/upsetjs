```jsx live
// import React from 'react';

function Example() {
  const [selection, setSelection] = useState(null);
  const elems = useMemo(
    () =>
      Array(100)
        .fill(0)
        .map(() => ({
          a: Math.random(),
          b: Math.random(),
        })),
    []
  );
  return (
    <Scatterplot
      selection={selection}
      onHover={setSelection}
      width={500}
      height={500}
      elems={elems}
      xAttr="a"
      yAttr="b"
      title="As"
    />
  );
}
```

function Example() {
