```jsx live
// import React from 'react';

function Example() {
  const [selection, setSelection] = useState<UpSetSelection<any>>(null);
  const elems = useMemo(() => Array(100)
    .fill(0)
    .map((_, i) => ({
      n: i.toString(),
      a: Math.random(),
    }), []);
  return (
    <Histogram selection={selection} onHover={setSelection} width={500} height={100} elems={elems} attr="a" title="As" />
  );
}
```
