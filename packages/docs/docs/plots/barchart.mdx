```jsx live
// import React from 'react';

function Example() {
  const [selection, setSelection] = useState<UpSetSelection<any>>(null);
  const elems = useMemo(() => Array(10)
    .fill(0)
    .map((_, i) => ({
      n: i.toString(),
      a: Math.random(),
    }), []);
  return (
    <BarChart width={500} height={100} selection={selection} onHover={setSelection} elems={elems} vAttr="a" iAttr="n" />
  );
}
```
